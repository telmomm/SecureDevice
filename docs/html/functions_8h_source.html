34float readBatteryLevel(int VBATPIN){
35 uint32_t raw = analogRead(VBATPIN);
36 float voltage = raw * 3.3 / 4095.0;
37 float maxVoltage = 3.6;
38 float minVoltage = 2.55;
39 float percentage = (voltage - minVoltage) / (maxVoltage - minVoltage) * 100;
40 return percentage;
41}
